struct window{
    char* name;
    int top;
    int left;
    int bottom;
    int right;
    char* handle;
}

void printWindows(Window *w, uint64 n){

  printDesktop(); // a function that will print the background

   for(uint64 i = 0; i < n; i++)
   {
      for(uint64 x = w[i].left; x < w[i].right; x++)
        for(uint64 y = w[i].top; y < w[i].bottom; y++)
        {
            putpixel(x, y, 15);
        }
   }
}

window Window(int y, int x, int ymax, intxmax, char* name){
    window w;
    w.name = name;
    w.top = y;
    w.left = x;
    w.bottom = ymax;
    w.right = xmax;
    w.handle = name;
    return w;
}
